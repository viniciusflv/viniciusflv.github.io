  commit id: "initial commit"

  branch hom
  branch feat/button

  checkout feat/button

    commit id: "finish button"

  checkout hom
    merge feat/button

  checkout feat/button
    commit id: "fix hover"

  checkout hom

  checkout feat/button
    commit id: "fix accessibility"

  checkout hom
    merge feat/button

  checkout main
    merge hom
